    float cxx = cov2[0][0] + 0.3;
    float cxy = cov2[1][0];
    float cyy = cov2[1][1] + 0.3;
    
    float det = cxx*cyy - cxy*cxy;
    if (det == 0.0) return;
    float inv_det = 1.0/det;
    v_conic = vec3(cyy*inv_det, -cxy*inv_det, cxx*inv_det);
    
    // Screen-space extent (3 sigma)
    float half_x = 3.0 * sqrt(cxx);
    float half_y = 3.0 * sqrt(cyy);
    
    // Clip-space position of splat center
    vec4 pos_clip = MCDCMatrix * gl_in[0].gl_Position;
    
    // Convert half-extents to NDC (need viewport dimensions)
    // NDC half-width for half_x pixels: half_x / (viewport_width/2) = 2*half_x/viewport_width
    // But we have focal_x = viewport_width / (2*tan_fovx)
    // So viewport_width = 2 * focal_x * tan_fovx
    float vp_width = 2.0 * focal_x * tan_fovx;
    float vp_height = 2.0 * focal_y * tan_fovy;
    float quad_x = 2.0 * half_x / vp_width;  // NDC units
    float quad_y = 2.0 * half_y / vp_height;  // NDC units
    
    // Need to scale by w for proper NDC
    float w = pos_clip.w;
    
    v_color = vs_color[0];
    v_opacity = vs_opacity_out[0];
    
    // Emit 4 vertices forming a quad
    vec2 offsets[4] = vec2[4](
        vec2(-1, -1), vec2(1, -1), vec2(-1, 1), vec2(1, 1)
    );
    
    for (int i = 0; i < 4; i++) {
        v_coordxy = vec2(offsets[i].x * half_x, offsets[i].y * half_y);
        gl_Position = vec4(
            pos_clip.x + offsets[i].x * quad_x * w,
            pos_clip.y + offsets[i].y * quad_y * w,
            pos_clip.z,
            w
        );
        EmitVertex();
    }
    EndPrimitive();
}
